import csv
import importlib.util
import json
import sys
from pathlib import Path

bundle = Path(sys.argv[1])
spec = importlib.util.spec_from_file_location("stage4_integrity", bundle / "stage4_integrity.py")
stage4 = importlib.util.module_from_spec(spec)
assert spec.loader is not None
spec.loader.exec_module(stage4)

rows = []
all_green = True
for probe_id in stage4.EXPECTED:
    log_path = bundle / f"{probe_id}.log"
    rc_path = bundle / f"{probe_id}.rc"
    rc = int(rc_path.read_text().strip())
    detected_pass, evidence = stage4.detect_from_log(probe_id, log_path.read_text())
    sidecar = {
        "probe_id": probe_id,
        "rc": rc,
        "pass": detected_pass,
        "detect_kind": stage4.DETECT_KIND[probe_id],
        "log_path": str(log_path),
        "parsed_evidence": evidence,
    }
    (bundle / f"{probe_id}.json").write_text(json.dumps(sidecar, indent=2, sort_keys=True) + "\n")
    rows.append({
        "probe_id": probe_id,
        "rc": str(rc),
        "pass": "1" if detected_pass else "0",
        "log_path": str(log_path),
    })
    all_green = all_green and rc == 0 and detected_pass

with (bundle / "probe_results.tsv").open("w", newline="") as handle:
    writer = csv.DictWriter(handle, fieldnames=["probe_id", "rc", "pass", "log_path"], delimiter="\t")
    writer.writeheader()
    writer.writerows(rows)

(bundle / "all_green.txt").write_text(("1" if all_green else "0") + "\n")
PY
