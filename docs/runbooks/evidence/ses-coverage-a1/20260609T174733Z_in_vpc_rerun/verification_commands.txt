import importlib.util
import json
import sys
from pathlib import Path

bundle = Path(sys.argv[1])
spec = importlib.util.spec_from_file_location("stage4_integrity", bundle / "stage4_integrity.py")

rows = ["probe_id\trc\tpass\tlog_path"]

for probe_id in stage4.EXPECTED:
    log_path = bundle / f"{probe_id}.log"
    rc = int((bundle / f"{probe_id}.rc").read_text().strip())
    passed, parsed_evidence = stage4.detect_from_log(probe_id, log_path.read_text())
    sidecar = {
        "detect_kind": stage4.DETECT_KIND[probe_id],
        "log_path": str(log_path),
        "parsed_evidence": parsed_evidence,
        "pass": passed,
        "probe_id": probe_id,
        "rc": rc,
    }
    (bundle / f"{probe_id}.json").write_text(json.dumps(sidecar, indent=2, sort_keys=True) + "\n")
    rows.append(f"{probe_id}\t{rc}\t{int(passed)}\t{log_path}")
    all_green = all_green and rc == 0 and passed
(bundle / "probe_results.tsv").write_text("\n".join(rows) + "\n")
